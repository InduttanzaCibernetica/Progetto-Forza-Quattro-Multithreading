  interface ServerEvent {
    +getId(): String
  }

package data {

  class Board {
    +isColumnAvailable(col: int): boolean
    +applyMove(token: Token, col: int): boolean
    +checkVictory(token: Token): boolean
    +isFull(): boolean
    +toLinearState(): String
  }

  class Player {
    +getId(): int
    +getName(): String
    +getAge(): int
    +getToken(): Token
    +setToken(token: Token): void
  }

  class PlayerSession {
    +getPlayer(): Player
    +getHandler(): ClientHandler
  }

  class PlayerQueue {
    +enqueue(session: PlayerSession): void
    +dequeue(): PlayerSession
    +isEmpty(): boolean
    +getSize(): int
  }

  class GameQueue {
    +addGame(game: GameRoom): void
    +removeGame(game: GameRoom): void
  }
}

package server {

  class Server {
    +start(): void
    +main(args: String[]): void
  }

  class MessageParser {
    +parse(msg: String): ClientCommand
  }

  class MessageFormatter {
    +format(ev: ServerEvent): String
  }

  class TurnTimer {
    +start(room: GameRoom, dueMillis: long): void
    +cancel(): void
  }

  class ClientHandler {
    +run(): void
    +handleMessage(raw: String): void
    +sendMessage(msg: String): void
    +setRoom(room: GameRoom): void
  }

  class GameManager {
    +assignPlayer(session: PlayerSession): void
    +createGame(p1: PlayerSession, p2: PlayerSession): void
    +chooseFirst(p1: PlayerSession, p2: PlayerSession): PlayerSession
  }

  class GameRoom {
    +start(): void
    +handleMove(session: PlayerSession, col: int): boolean
    +onTimeout(): void
    +onDisconnect(session: PlayerSession): void
  }
}

Server --> ClientHandler : crea *
Server --> GameManager : usa
ClientHandler --> GameManager : usa
GameManager --> GameRoom : crea
GameRoom --> Board : contiene
GameRoom --> PlayerSession : p1/p2
GameRoom ..> TurnTimer : usa
ClientHandler ..> MessageParser : parsing
ClientHandler ..> MessageFormatter : serializza
MessageParser ..> ClientCommand
MessageFormatter ..> ServerEvent
